27void Tournament::calculate_rankings(const BinTree<match>& t, const vector<int>& rank_scores, vector<int>& p, int k) {
28 int i = t.value().a-1;
29 int j = t.value().b-1;
30 int n = t.value().result.size();
31 // Solo añadimos los juegos y sets ganados y perdidos cuando se han
32 // disputado un mínimo de 2 sets, es decir, que el partido se ha disputado
33 if (n > 1) {
34 for (int s = 0; s < n; ++s) {
35 stats[i].set_games_won(stats[i].get_games_won() + t.value().result[s].first);
36 stats[i].set_games_lost(stats[i].get_games_lost() + t.value().result[s].second);
37 stats[j].set_games_won(stats[j].get_games_won() + t.value().result[s].second);
38 stats[j].set_games_lost(stats[j].get_games_lost() + t.value().result[s].first);
39 if (t.value().result[s].first > t.value().result[s].second) {
40 stats[i].set_sets_won(stats[i].get_sets_won() + 1);
41 stats[j].set_sets_lost(stats[j].get_sets_lost() + 1);
42 }
43 else {
44 stats[j].set_sets_won(stats[j].get_sets_won() + 1);
45 stats[i].set_sets_lost(stats[i].get_sets_lost() + 1);
46 }
47 }
48 }
49 if (first_wins(t.value())) {
50 stats[i].set_matches_won(stats[i].get_matches_won() + 1);
51 stats[j].set_matches_lost(stats[j].get_matches_lost() + 1);
52 if (k < p[i] or p[i] == 0) {
53 p[i] = k;
54 stats[i].set_points(rank_scores[k-1]);
55 }
56 if (k+1 < p[j] or p[j] == 0) {
57 p[j] = k+1;
58 stats[j].set_points(rank_scores[k]);
59 }
60 }
61 else {
62 stats[j].set_matches_won(stats[j].get_matches_won() + 1);
63 stats[i].set_matches_lost(stats[i].get_matches_lost() + 1);
64 if (k+1 < p[i] or p[i] == 0) {
65 p[i] = k+1;
66 stats[i].set_points(rank_scores[k]);
67 }
68 if (k < p[j] or p[j] == 0) {
69 p[j] = k;
70 stats[j].set_points(rank_scores[k-1]);
71 }
72 }
73 if (not t.left().empty()) calculate_rankings(t.left(), rank_scores, p, k+1);
74 if (not t.right().empty()) calculate_rankings(t.right(), rank_scores, p, k+1);
