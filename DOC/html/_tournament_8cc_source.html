96void Tournament::calculate_rankings(const BinTree<match>& t, const vector<int>& rank_scores, vector<int>& p, int k) {
97 int i = t.value().a-1;

102 if (n > 1) {
103 for (int s = 0; s < n; ++s) {
104 stats[i].set_games_won(stats[i].get_games_won() + t.value().result[s].first);
105 stats[i].set_games_lost(stats[i].get_games_lost() + t.value().result[s].second);
106 stats[j].set_games_won(stats[j].get_games_won() + t.value().result[s].second);
107 stats[j].set_games_lost(stats[j].get_games_lost() + t.value().result[s].first);
108 if (t.value().result[s].first > t.value().result[s].second) {
109 stats[i].set_sets_won(stats[i].get_sets_won() + 1);
110 stats[j].set_sets_lost(stats[j].get_sets_lost() + 1);
111 }
112 else {
113 stats[j].set_sets_won(stats[j].get_sets_won() + 1);
114 stats[i].set_sets_lost(stats[i].get_sets_lost() + 1);
115 }
116 }
117 }
118 if (first_wins(t.value())) {
119 stats[i].set_matches_won(stats[i].get_matches_won() + 1);
120 stats[j].set_matches_lost(stats[j].get_matches_lost() + 1);
121 if (k < p[i] or p[i] == 0) {
122 p[i] = k;
123 stats[i].set_points(rank_scores[k-1]);
